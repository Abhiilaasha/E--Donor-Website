CREATE TABLE registration ( REGISTRATION_ID INT AUTO_INCREMENT PRIMARY KEY, FIRST_NAME VARCHAR(50) NOT NULL, LAST_NAME VARCHAR(50) NOT NULL, Age INT NOT NULL, BLOOD_GROUPr VARCHAR(5) NOT NULL, Gender VARCHAR(10) NOT NULL, EMAIL VARCHAR(100) UNIQUE NOT NULL, PASS_WORD VARCHAR(255) NOT NULL, phoneR VARCHAR(20) NOT NULL, ADD_RESSr VARCHAR(255) NOT NULL, ZIP_CODE VARCHAR(10) NOT NULL );




2.)CREATE TABLE organ_requests ( REQUEST_ID INT AUTO_INCREMENT PRIMARY KEY, REGISTRATION_ID INT, ORGAN_TYPE VARCHAR(50), QUANTITY INT, BLOOD_GROUPn VARCHAR(10), REQUEST_TYPE VARCHAR(50), ADD_RESSn VARCHAR(255), ZIP_CODE VARCHAR(10), phoneN VARCHAR(15), REQUEST_TIME TIMESTAMP DEFAULT CURRENT_TIMESTAMP, FOREIGN KEY (REGISTRATION_ID) REFERENCES registration(REGISTRATION_ID) );




3.)CREATE TABLE donation_record ( DONATION_ID INT AUTO_INCREMENT PRIMARY KEY, REGISTRATION_ID INT, ORGAN_TYPEd VARCHAR(50), BLOOD_GROUPd VARCHAR(5), QUANTITYd INT, short_note TEXT, FOREIGN KEY (REGISTRATION_ID) REFERENCES registration(REGISTRATION_ID) );
